### Código dos endpoints — app/routers/microsoft_marketplace.py

```python
from fastapi import APIRouter, Request, BackgroundTasks
from fastapi.responses import RedirectResponse
from src.database.supabase_client import SupabaseClient
from src.monetization.stripe_client import StripeClient
import logging

router = APIRouter(prefix="/api/microsoft", tags=["microsoft"])
logger = logging.getLogger(__name__)

@router.get("/landing")
async def landing_page(token: str, request: Request):
    """
    Microsoft redireciona aqui após o cliente assinar pelo marketplace.
    Token contém subscription_id e dados do cliente.
    """
    # Salva lead no Supabase
    db = SupabaseClient()
    await db.table("microsoft_subscriptions").insert({
        "token": token,
        "status": "pending_activation",
        "source": "microsoft_marketplace"
    }).execute()

    # Redireciona para onboarding
    return RedirectResponse(
        url=f"https://global-engenharia.com/ecosystem/activate?token={token}"
    )

@router.post("/webhook")
async def fulfillment_webhook(
    request: Request,
    background_tasks: BackgroundTasks
):
    """
    Microsoft notifica mudanças de assinatura:
    - Subscribed: novo cliente
    - Unsubscribed: cancelamento
    - Suspended: pagamento falhou
    - Reinstated: pagamento recuperado
    """
    data = await request.json()
    action = data.get("action", "")
    subscription_id = data.get("subscriptionId", "")

    logger.info(f"Microsoft webhook: {action} | {subscription_id}")

    background_tasks.add_task(
        process_microsoft_subscription,
        action, subscription_id, data
    )
    return {"status": "ok"}

async def process_microsoft_subscription(
    action: str,
    subscription_id: str,
    data: dict
):
    db = SupabaseClient()
    if action == "Subscribed":
        await db.table("tenants").insert({
            "source": "microsoft_marketplace",
            "subscription_id": subscription_id,
            "plan": data.get("planId"),
            "status": "active",
            "company": data.get("purchaser", {}).get("emailId", "")
        }).execute()
        logger.info(f"Novo cliente Microsoft ativado: {subscription_id}")

    elif action in ["Unsubscribed", "Suspended"]:
        await db.table("tenants").update(
            {"status": action.lower()}
        ).eq("subscription_id", subscription_id).execute()
        logger.info(f"Assinatura {action}: {subscription_id}")

    elif action == "Reinstated":
        await db.table("tenants").update(
            {"status": "active"}
        ).eq("subscription_id", subscription_id).execute()
```

### Tabela Supabase

```sql
CREATE TABLE microsoft_subscriptions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    subscription_id TEXT UNIQUE,
    token TEXT,
    plan TEXT,
    status TEXT DEFAULT 'pending',
    company TEXT,
    email TEXT,
    source TEXT DEFAULT 'microsoft_marketplace',
    activated_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_ms_sub_id ON microsoft_subscriptions(subscription_id);
CREATE INDEX idx_ms_status ON microsoft_subscriptions(status);
```
